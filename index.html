let countdownTime = 259200; const startTimeKey = "treeGrowthStartTime"; function startGrowthTimer() { // Check if start time is already stored let startTime = localStorage.getItem(startTimeKey); if (!startTime) { // Store current time as start time startTime = Date.now(); localStorage.setItem(startTimeKey, startTime); } // Calculate elapsed time const elapsedTime = Math.floor((Date.now() - startTime) / 1000); const remainingTime = countdownTime - elapsedTime; // If time's up, reset and show "Time's up" if (remainingTime <= 0) { document.querySelector('.coconut-tree').style.transform = "scaleY(1)"; document.getElementById("countdown-timer").textContent = "Time's up!"; localStorage.removeItem(startTimeKey); return; } // Update countdown display updateCountdown(remainingTime); // Set animation duration based on remaining time document.querySelector('.coconut-tree').style.animationDuration = `${remainingTime}s`; } function updateCountdown(remainingTime) { const countdownElement = document.getElementById("countdown-timer"); const days = Math.floor(remainingTime / (24 * 3600)); const hours = Math.floor((remainingTime % (24 * 3600)) / 3600); const minutes = Math.floor((remainingTime % 3600) / 60); const seconds = remainingTime % 60; const formattedTime = String(days).padStart(2, '0') + "d " + String(hours).padStart(2, '0') + "h:" + String(minutes).padStart(2, '0') + "m:" + String(seconds).padStart(2, '0') + "s"; countdownElement.textContent = formattedTime; } // Initialize the growth timer startGrowthTimer();
